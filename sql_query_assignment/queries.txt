-- Creating a table
CREATE TABLE Students(
 id INT Primary Key,
 name VARCHAR(50),
 score INT,
 passed BOOLEAN,
 class INT(5)
);

-- Retriving all columns 
SELECT *
FROM Students;

-- Retrieve only name and score
SELECT name,score
FROM Students

-- Students scored above 80
SELECT *
FROM Students
WHERE score>80;

-- Students scored above 80 and passed
SELECT *
FROM Students
WHERE score>80 AND passed=TRUE;

-- Students score above 85 or class is 9
SELECT *
FROM Students
WHERE score>85 OR class=9;

-- Students who did not pass exam
SELECT *
FROM Students
WHERE passed=FALSE;

-- List of distinct classes
SELECT DISTINCT class
FROM Students;

-- Top five highest scorer
SELECT *
FROM Students
ORDER BY score DESC
LIMIT 5;

-- sort students first by class and then by score
SELECT *
FROM Students
ORDER BY class ASC,score DESC;

-- alias score as final_score
SELECT name,score AS final_score
FROM Students;

-- top 3 student scored abobe 75 sorted in descending ORDER
SELECT name,score 
FROM Students
WHERE score>75
ORDER BY score DESC
LIMIT 3;
